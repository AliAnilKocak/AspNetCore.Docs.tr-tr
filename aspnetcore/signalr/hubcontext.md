### <a name="inject-a-strongly-typed-hubcontext"></a><span data-ttu-id="8245d-124">Türü kesin belirlenmiş bir HubContext Ekle</span><span class="sxs-lookup"><span data-stu-id="8245d-124">Inject a strongly-typed HubContext</span></span>

```csharp
{
    public IHubContext<ChatHub, IChatClient> _strongChatHubContext { get; }

    public ChatController(IHubContext<ChatHub, IChatClient> chatHubContext)
    {
        _strongChatHubContext = chatHubContext;
    }

    public async Task SendMessage(string message)
    {
        await _strongChatHubContext.Clients.All.ReceiveMessage(message);
    }
}
```
